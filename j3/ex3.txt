FONCTION calculerMoyenne(note1, note2, note3)
    RETOURNER note1 + note2 + note3 / 3
FIN

FONCTION attribuerMention(moyenne)
    SI moyenne >= 16
        RETOURNER 'Très bien'
    SINON SI moyenne >= 14
        RETOURNER 'Bien'
    SINON SI moyenne >= 12
        RETOURNER 'Assez bien'
    SINON SI moyenne >= 10
        RETOURNER 'Passable'
    SINON SI moyenne < 10
        RETOURNER 'Insuffisant'
    FIN SI
FIN

Demander 1e note
x <- 1e note
Demander 2e note
y <- 2e note
Demander 3e note
z <- 3e note

moyenne <- calculerMoyenne(x, y, z)
mention <- attribuerMention(moyenne)

AFFICHER 'Moyenne : ' + moyenne + '</br>Mention : ' + mention
