//Exercice 1 : Afficher une grille de morpion
//Créer un tableau 3×3 représentant une grille de morpion.
// Initialiser la grille avec des espaces vides (" ")

tab <- [["","",""],["","",""],["","",""]]

// Placer quelques coups (X et O)

tab[3][2] <- "X"
tab[0][1] <- "O"
tab[3][1] <- "X"
tab[0][0] <- "O"

// Afficher la grille de manière lisible

POUR i de 0 à taille de tab - 1
    AFFICHER tab[i]
FIN POUR
